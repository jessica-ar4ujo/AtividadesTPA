package lista3;

public class Altura {

    public static void main(String[] args) {
        double joao = 1.34;
        double pedro = 1.45;
        int anos = 0;

        while (joao < pedro) {
            joao += 0.025; 
            pedro += 0.020; 
            anos++;
        }

        System.out.println("João ficará mais alto que Pedro em: " + anos + " anos.");
    }
}
